currentTrip structure:
{
  tripId: string - X
  days: integer  - X
  startDate: daysJs
  tripName: string - X
  headerImgUrl: string - X
  locationData: object
  itinerary: [{},{}]
}

locationData structure:
{
  country: string
  country_code: string
  latitude: float
  longitude: float
  name: string
  place: string
}

itinerary structure:
[{}, {}, {}]

{

  dayStartTime: string,

  route: TBC

{
